     [0,cos(psi),-sin(psi)],
     [0,sin(psi),cos(psi)]])

    [[cos(theta),0,sin(theta)],
     [0,1,0],
     [-sin(theta),0,cos(theta)]])

    [[cos(phi),-sin(phi),0],
     [sin(phi),cos(phi),0],

#final orientation of the frame.
final_orientation=simplify(R_phi*R_theta*R_psi)
